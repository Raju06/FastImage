105  void executeTask
106  (std::shared_ptr<fi::HTGSTileRequestData<UserType>> tileRequestData) final {
107  CachedTile<UserType> *cachedTile;
108  uint32_t row = tileRequestData->getIndexRowTileAsked();
109  uint32_t col = tileRequestData->getIndexColTileAsked();
110 
111  // Get locked tile from the cache, can be empty or not
112  cachedTile = _cache->getLockedTile(row, col);
113 
114  // Set the tile if empty
115  if (cachedTile->isNewTile()) {
116  cachedTile->setNewTile(false);
117  _cache->addTimeDisk(loadTileFromFile(cachedTile->getData(), row, col));
118  }
119 
120  // Copy the tile or part of it into the view
121  copyTileToView(tileRequestData, cachedTile);
122  cachedTile->unlock();
123 
124  htgs::m_data_t<View<UserType>> viewData = tileRequestData->getViewData();
125  this->addResult(tileRequestData);
126  }

131  void copyTileToView(
132  std::shared_ptr<fi::HTGSTileRequestData<UserType>> tileRequestData,
133  CachedTile<UserType> *cachedTile) {
134  uint32_t
135  rowFrom = tileRequestData->getRowFrom(),
136  colFrom = tileRequestData->getColFrom(),
137  rowDest = tileRequestData->getRowDest(),
138  colDest = tileRequestData->getColDest(),
139  heightToCopy = tileRequestData->getHeightToCopy(),
140  widthToCopy = tileRequestData->getWidthToCopy(),
141  tileWidth = tileRequestData->getTileWidth(),
142  viewWidth = tileRequestData->getViewWidth();
143 
144  View<UserType> *dest = tileRequestData->getViewData()->get();
145  CachedTile<UserType> *src = cachedTile;
146 
147  for (uint32_t r = 0; r < heightToCopy; ++r) {
148  std::copy_n(
149  src->getData() + ((rowFrom + r) * tileWidth + colFrom),
150  widthToCopy,
151  dest->getData() + ((rowDest + r) * viewWidth + colDest)
152  );
153  }
154  }
