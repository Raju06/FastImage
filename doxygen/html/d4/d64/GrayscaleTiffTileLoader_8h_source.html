95  // Open the file
96  this->_tiff = TIFFOpen(fileName.c_str(), "r");
97  if (this->_tiff != nullptr) {
98  if (TIFFIsTiled(_tiff) == 0) {
99  std::stringstream message;
100  message << "Tile Loader ERROR: The image is not tiled.";
101  std::string m = message.str();
102  throw (FastImageException(m));
103  }
104  // Load/parse header
105  TIFFGetField(this->_tiff, TIFFTAG_IMAGEWIDTH, &(this->_imageWidth));
106  TIFFGetField(this->_tiff, TIFFTAG_IMAGELENGTH, &(this->_imageHeight));
107  TIFFGetField(this->_tiff, TIFFTAG_TILEWIDTH, &this->_tileWidth);
108  TIFFGetField(this->_tiff, TIFFTAG_TILELENGTH, &this->_tileHeight);
109  TIFFGetField(this->_tiff, TIFFTAG_SAMPLESPERPIXEL, &samplesPerPixel);
110  TIFFGetField(this->_tiff, TIFFTAG_BITSPERSAMPLE, &(this->_bitsPerSample));
111  TIFFGetField(this->_tiff, TIFFTAG_SAMPLEFORMAT, &(this->_sampleFormat));
112 
113  // Test if the file is greyscale
114  if (samplesPerPixel != 1) {
115  std::stringstream message;
116  message
117  << "Tile Loader ERROR: The image is not greyscale: "
118  "SamplesPerPixel = " << samplesPerPixel << ".";
119  std::string m = message.str();
120  throw (FastImageException(m));
121  }
122 
123  // Interpret undefined data format as unsigned integer data
124  if (_sampleFormat < 1 && _sampleFormat > 3) {
125  _sampleFormat = 1;
126  }
127 
128  } else {
129  std::stringstream message;
130  message << "Tile Loader ERROR: The image can not be opened.";
131  std::string m = message.str();
132  throw (FastImageException(m));
133  }

213  double loadTileFromFile(UserType *tile,
214  uint32_t indexRowGlobalTile,
215  uint32_t indexColGlobalTile) override {
216  tdata_t tiffTile = nullptr;
217  tiffTile = _TIFFmalloc(TIFFTileSize(_tiff));
218  auto begin = std::chrono::high_resolution_clock::now();
219  TIFFReadTile(_tiff,
220  tiffTile,
221  indexColGlobalTile * _tileWidth,
222  indexRowGlobalTile * _tileHeight,
223  0,
224  0);
225  auto end = std::chrono::high_resolution_clock::now();
226  double diskDuration = (std::chrono::duration_cast<std::chrono::nanoseconds>(
227  end - begin).count());
228  std::stringstream message;
229  switch (this->_sampleFormat) {
230  case 1 :
231  switch (this->_bitsPerSample) {
232  case 8:loadTile<uint8_t>(tiffTile, tile);
233  break;
234  case 16:loadTile<uint16_t>(tiffTile, tile);
235  break;
236  case 32:loadTile<uint32_t>(tiffTile, tile);
237  break;
238  case 64:loadTile<uint64_t>(tiffTile, tile);
239  break;
240  default:
241  message
242  << "Tile Loader ERROR: The data format is not supported for "
243  "unsigned integer, number bits per pixel = "
244  << this->_bitsPerSample;
245  std::string m = message.str();
246  throw (FastImageException(m));
247  }
248  break;
249  case 2:
250  switch (this->_bitsPerSample) {
251  case 8:loadTile<int8_t>(tiffTile, tile);
252  break;
253  case 16:loadTile<int16_t>(tiffTile, tile);
254  break;
255  case 32:loadTile<int32_t>(tiffTile, tile);
256  break;
257  case 64:loadTile<int64_t>(tiffTile, tile);
258  break;
259  default:
260  message
261  << "Tile Loader ERROR: The data format is not supported for "
262  "signed integer, number bits per pixel = "
263  << this->_bitsPerSample;
264  std::string m = message.str();
265  throw (FastImageException(m));
266  }
267  break;
268  case 3:
269  switch (this->_bitsPerSample) {
270  case 8:loadTile<float>(tiffTile, tile);
271  break;
272  case 16:loadTile<float>(tiffTile, tile);
273  break;
274  case 32:loadTile<float>(tiffTile, tile);
275  break;
276  case 64:loadTile<double>(tiffTile, tile);
277  break;
278  default:
279  message
280  << "Tile Loader ERROR: The data format is not supported for "
281  "float, number bits per pixel = " << this->_bitsPerSample;
282  std::string m = message.str();
283  throw (FastImageException(m));
284  }
285  break;
286  default:
287  message
288  << "Tile Loader ERROR: The data format is not supported, sample "
289  "format = " << this->_sampleFormat;
290  std::string m = message.str();
291  throw (FastImageException(m));
292  }
293 
294  _TIFFfree(tiffTile);
295  return diskDuration;
296  }
