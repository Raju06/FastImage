84  void executeTask(
85  std::shared_ptr<fi::HTGSViewRequestData<UserType>> viewRequest) {
86  if (_nbReleasePyramid[this->getPipelineId()] == 0) {
87  return;
88  }
89  htgs::m_data_t<View<UserType>> viewMemory =
90  ViewLoader<UserType>::template
91  getMemory<View<
92  UserType>>(
93  "viewMem",
94  new ReleaseCountRule(_nbReleasePyramid[this->getPipelineId()]));
95  viewMemory->get()->init(viewRequest);
96  uint32_t
97  tileHeight = viewRequest->getTileHeight(),
98  tileWidth = viewRequest->getTileWidth(),
99  minRowFile = viewRequest->getMinRowFile(),
100  minColFile = viewRequest->getMinColFile(),
101  maxRowFile = viewRequest->getMaxRowFile(),
102  maxColFile = viewRequest->getMaxColFile(),
103  rowAlreadyFilled = viewRequest->getTopFill(),
104  colAlreadyFilled = 0,
105  topFill = viewRequest->getTopFill(),
106  leftFill = viewRequest->getLeftFill(),
107  botFill = viewRequest->getBottomFill(),
108  rightFill = viewRequest->getRightFill(),
109  rowFrom = 0,
110  colFrom = 0,
111  heightToCopy = 0,
112  widthToCopy = 0,
113  ulTileRowGlobal = 0,
114  ulTileColGlobal = 0;
115 
116  for (uint32_t r = viewRequest->getIndexRowMinTile();
117  r < viewRequest->getIndexRowMaxTile(); r++) {
118  ulTileRowGlobal = r * tileHeight;
119  rowFrom =
120  ulTileRowGlobal <= minRowFile ? (minRowFile - ulTileRowGlobal) : 0;
121  heightToCopy =
122  std::min(maxRowFile, ulTileRowGlobal + tileHeight) - rowFrom
123  - ulTileRowGlobal;
124 
125  colAlreadyFilled = leftFill;
126  for (uint32_t c = viewRequest->getIndexColMinTile();
127  c < viewRequest->getIndexColMaxTile(); c++) {
128  ulTileColGlobal = c * tileWidth;
129  colFrom =
130  ulTileColGlobal <= minColFile ? (minColFile - ulTileColGlobal) : 0;
131  widthToCopy =
132  std::min(maxColFile, ulTileColGlobal + tileWidth) - colFrom
133  - ulTileColGlobal;
134 
135  // Create the tile request
136  auto htgsTileRequestData =
137  new fi::HTGSTileRequestData<UserType>(r, c, viewMemory,
138  viewRequest);
139  htgsTileRequestData->setRowFrom(rowFrom);
140  htgsTileRequestData->setColFrom(colFrom);
141  htgsTileRequestData->setRowDest(rowAlreadyFilled);
142  htgsTileRequestData->setColDest(colAlreadyFilled);
143  htgsTileRequestData->setHeightToCopy(heightToCopy);
144  htgsTileRequestData->setWidthToCopy(widthToCopy);
145  htgsTileRequestData->setTopToFill(topFill);
146  htgsTileRequestData->setRightToFill(rightFill);
147  htgsTileRequestData->setBottomToFill(botFill);
148  htgsTileRequestData->setLeftToFill(leftFill);
149  this->addResult(htgsTileRequestData);
150  colAlreadyFilled += widthToCopy;
151  }
152  rowAlreadyFilled += heightToCopy;
153  }
154  }
