192  _minRowCenterTileGlobal = viewRequest->getIndexRowCenterTile() * tileHeight;
193  _minColCenterTileGlobal = viewRequest->getIndexColCenterTile() * tileWidth;
194  _maxRowCenterTileGlobal = std::min(viewRequest->getImageHeight(),
195  _minRowCenterTileGlobal + tileHeight);

197  _minColCenterTileGlobal + tileWidth);
198  _maxColCenterTileLocal = _maxColCenterTileGlobal

200  _maxRowCenterTileLocal = _maxRowCenterTileGlobal
201  - viewRequest->getIndexRowCenterTile() * tileHeight;
