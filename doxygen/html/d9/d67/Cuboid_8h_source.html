201 template<class T>
202 T
203 Cuboid<T>::getDistanceSqrTo(Vector2<T> const &point) const {
204  T distanceSqr = 0;
205 
206  for (int dim = 0; dim < 2; ++dim) {
207  T coordDistance = std::max(_minCoords[dim] - point[dim],
208  point[dim] - _maxCoords[dim]);
209 
210  coordDistance = std::max(coordDistance, (T) 0);
211 
212  distanceSqr += pow(coordDistance, 2);
213  }
214 
215  return distanceSqr;
216 }

220 template<class T>
221 Vector2<T>
222 Cuboid<T>::getFarthestPoint(Vector2<T> const &queryPoint) const {
223  Vector2<T> farthestPoint;
224 
225  Vector2<T> center = (_minCoords + _maxCoords) * 0.5;
226 
227  for (int dim = 0; dim < 2; ++dim) {
228  if (queryPoint[dim] < center[dim]) {
229  farthestPoint[dim] = _maxCoords[dim];
230  } else {
231  farthestPoint[dim] = _minCoords[dim];
232  }
233  }
234 
235  return farthestPoint;
236 }
