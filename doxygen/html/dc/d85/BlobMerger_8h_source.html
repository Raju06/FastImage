121  void merge() {
122  UnionFind<Blob>
123  uf{};
124 
125  std::map<Blob *, std::set<Blob *>>
126  parentSons{};
127 
128  // Apply the UF algorithm to every linked blob
129  for (auto blobCoords : _toMerge) {
130  for (auto coord : blobCoords.second) {
131  uf.unionElements(blobCoords.first,
132  getBlobFromCoord(coord.first, coord.second));
133  }
134  }
135 
136  // Clear to merge data stucture
137  _toMerge.clear();
138 
139  // Associate every blob to it parent
140  for (auto blob : _blobs->_blobs) {
141  parentSons[uf.find(blob)].insert(blob);
142  }
143 
144  // For every Parent - son blob, merge every parent to it son
145  for (auto pS : parentSons) {
146  auto sons = pS.second;
147  Blob
148  *toMerge = *sons.begin(),
149  *merged = nullptr;
150  for (auto son = std::next(sons.begin()); son != sons.end(); ++son) {
151  merged = toMerge->mergeAndDelete(*son);
152  if (merged == *son) {
153  _blobs->_blobs.remove(toMerge);
154  } else {
155  _blobs->_blobs.remove(*son);
156  }
157  toMerge = merged;
158  }
159  }
