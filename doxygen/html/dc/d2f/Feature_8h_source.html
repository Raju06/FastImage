168  bool isInBitMask(uint32_t row, uint32_t col) const {
169  bool answer = false;
170  if (contains(row, col)) {
171  uint32_t
172  // Find the position in the bounding box
173  localRow =
174  (uint32_t) (row - this->getBoundingBox().getUpperLeftRow()),
175  localCol =
176  (uint32_t) (col - this->getBoundingBox().getUpperLeftCol()),
177  absolutePosition =
178  localRow * this->getBoundingBox().getWidth() + localCol,
179  // Find the good "word" (uint32_t)
180  wordPosition = absolutePosition >> (uint32_t) 5,
181  // Find the good bit in the word
182  bitPosition =
183  (uint32_t) abs((int32_t) 32 - ((int32_t) absolutePosition
184  - (int32_t) (wordPosition << (uint32_t) 5)));
185  // Test if the bit is one
186  answer = (((((uint32_t) 1 << (bitPosition - (uint32_t) 1))
187  & this->_bitMask[wordPosition])
188  >> (bitPosition - (uint32_t) 1)) & (uint32_t) 1) == (uint32_t) 1;
189  }
190  return answer;
191  }
