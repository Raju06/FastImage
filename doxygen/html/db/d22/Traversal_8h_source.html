246  void d2xy(uint32_t n, uint32_t d, uint32_t &x, uint32_t &y) {
247  uint32_t rx, ry, s, t = d;
248  x = y = 0;
249  for (s = 1; s < n; s *= 2) {
250  rx = (uint32_t) 1 & (t / (uint32_t) 2);
251  ry = (uint32_t) 1 & (t ^ rx);
252  rot(s, x, y, rx, ry);
253  x += s * rx;
254  y += s * ry;
255  t /= 4;
256  }
257  }

340  pos = 0,
341  rowTemp = 0,
342  colTemp = 0,
343  minSize = (uint32_t) pow(2,
344  (uint32_t) std::min(log2(_numTileRow),

351  d2xy(minSize, pos, rowTemp, colTemp);

353  _traversal[pos].second = colTemp;
354  if (_traversal[pos].first > minX) minX = _traversal[pos].first;
355  if (_traversal[pos].second > minY) minY = _traversal[pos].second;
356  pos += 1;
357  }

359  // Filling empty case

361  if (row % 2 == 0) {
362  for (uint32_t col = minY + 1; col < _numTileCol; ++col) {
363  _traversal[pos].first = row;
364  _traversal[pos].second = col;
365  pos += 1;

369  _traversal[pos].first = row;
370  _traversal[pos].second = col;
371  pos += 1;

393  void spiralAlgorithm() {
394  static std::map<Direction, Direction> nextDirection = {
395  {Direction::EAST, Direction::SOUTH},
396  {Direction::SOUTH, Direction::WEST},
397  {Direction::WEST, Direction::NORTH},
398  {Direction::NORTH, Direction::EAST}};
399 
400  std::vector<std::vector<bool>>
401  hasNotBeenVisited(_numTileRow, std::vector<bool>(_numTileCol, true));
402  Direction
403  direction = Direction::EAST;
404 
405  uint32_t
406  row = 0,
407  col = 0,
408  step = 0;
409  int32_t
410  rowNext = 0,
411  colNext = 0;
412 
413  _traversal.emplace_back((uint32_t) row, (uint32_t) col);
414  hasNotBeenVisited[row][col] = false;
415 
416  while (step < _numTileRow * _numTileCol - 1) {
417  nextStep(direction, row, col, rowNext, colNext);
418  if (isFree(rowNext,
419  colNext,
420  _numTileRow,
421  _numTileCol,
422  hasNotBeenVisited)) {
423  ++step;
424  row = (uint32_t) rowNext;
425  col = (uint32_t) colNext;
426  hasNotBeenVisited[row][col] = false;
427  _traversal.emplace_back(row, col);
428  } else {
429  direction = nextDirection[direction];
430  }
431  }
432  }
