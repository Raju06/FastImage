83  void executeTask(
84  std::shared_ptr<fi::HTGSTileRequestData<UserType>> tileRequestData) {
85  auto data = tileRequestData->getViewData();
86 
87  // Tile without radius
88  if (tileRequestData->getViewRequest()->getNumberTilesToLoad() == 1 ||
89  tileRequestData->getViewRequest()->getRadius() == 0) {
90  fillGhostRegion(tileRequestData);
91  dataReady(data);
92  //this->addResult(data);
93  } else {
94  // Tile with radius
95  auto itPos = _countMap.find(data);
96  if (itPos != _countMap.end()) {
97  // Found pointer, so increment
98  (*itPos).second++;
99  auto val = (*itPos).second;
100  // If all the tiles have been collected for the view, then the view is
101  // complete
102  if (val == tileRequestData->getViewRequest()->getNumberTilesToLoad()) {
103  _countMap.erase(data);
104  fillGhostRegion(tileRequestData);
105  dataReady(data);
106  //this->addResult(data);
107  }
108  } else {
109  // Pointer was not found, so add it in with initial count of 1
110  _countMap.
111  insert(std::pair<htgs::m_data_t<View<UserType>>, uint32_t>(data, 1));
112  }
113  }

133  void fill(
134  std::shared_ptr<fi::HTGSTileRequestData<UserType>> tileRequestData) {
135 
136  UserType *tile = tileRequestData->getViewData()->get()->getData();
137 
138  uint32_t
139  topFill = tileRequestData->getTopToFill(),
140  bottomFill = tileRequestData->getBottomToFill(),
141  leftFill = tileRequestData->getLeftToFill(),
142  rightFill = tileRequestData->getRightToFill(),
143  viewHeight = tileRequestData->getViewHeight(),
144  viewWidth = tileRequestData->getViewWidth();
145 
146  for (uint32_t row = topFill; row < viewHeight - bottomFill; ++row) {
147  // L
148  std::fill_n(
149  tile + (row * viewWidth),
150  leftFill,
151  tile[row * viewWidth + leftFill]);
152  // R
153  std::fill_n(
154  tile + (row * viewWidth + viewWidth - rightFill),
155  rightFill,
156  tile[row * viewWidth + viewWidth - rightFill - 1]);
157  }
158  for (uint32_t row = 0; row < topFill; ++row) {
159  // UL
160  std::fill_n(
161  tile + (row * viewWidth),
162  leftFill,
163  tile[topFill * viewWidth + leftFill]);
164  // U
165  std::copy_n(
166  tile + (topFill * viewWidth + leftFill),
167  viewWidth - leftFill - rightFill,
168  tile + (row * viewWidth + leftFill));
169  // UR
170  std::fill_n(
171  tile + (row * viewWidth + viewWidth - rightFill),
172  rightFill,
173  tile[row * viewWidth + viewWidth - rightFill - 1]);
174  }
175  for (uint32_t row = viewHeight - bottomFill; row < viewHeight; ++row) {
176  // BL
177  std::fill_n(
178  tile + (row * viewWidth),
179  leftFill,
180  tile[(viewHeight - bottomFill - 1) * viewWidth + leftFill]);
181  // B
182  std::copy_n(
183  tile + ((viewHeight - bottomFill - 1) * viewWidth + leftFill),
184  viewWidth - leftFill - rightFill,
185  tile + (row * viewWidth + leftFill));
186  // BR
187  std::fill_n(
188  tile + (row * viewWidth + viewWidth - rightFill),
189  rightFill,
190  tile[(viewHeight - bottomFill) * viewWidth - rightFill - 1]);
191  }
192  }
