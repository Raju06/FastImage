69  HTGSViewRequestData(
70  uint32_t indexTileRow, uint32_t indexTileCol,
71  uint32_t numTilesHeight, uint32_t numTilesWidth, uint32_t radius,
72  uint32_t tileHeight, uint32_t tileWidth,
73  uint32_t imageHeight, uint32_t imageWidth, uint32_t level)
74  : _imageWidth(imageWidth),
75  _imageHeight(imageHeight),
76  _tileHeight(tileHeight),
77  _tileWidth(tileWidth),
78  _radius(radius),
79  _indexRowCenterTile(indexTileRow),
80  _indexColCenterTile(indexTileCol),
81  _level(level) {
82  // View Size
83  _viewHeight = tileHeight + 2 * radius;
84  _viewWidth = tileWidth + 2 * radius;
85  // Top left central tile pixel
86  _minRowCentralTile = _indexRowCenterTile * tileHeight;
87  _minColCentralTile = _indexColCenterTile * tileWidth;
88  // Index of the tile overlapped by the view.
89  _indexRowMinTile = (uint32_t) std::max(
90  (int32_t) _indexRowCenterTile
91  - (int32_t) ceil((double) radius / tileHeight),
92  (int32_t) 0);
93  _indexColMinTile = (uint32_t) std::max(
94  (int32_t) _indexColCenterTile
95  - (int32_t) ceil((double) radius / tileWidth),
96  (int32_t) 0);
97  _indexRowMaxTile = std::min(
98  _indexRowCenterTile + (int32_t) ceil((double) radius / tileHeight) + 1,
99  numTilesHeight);
100  _indexColMaxTile = std::min(
101  _indexColCenterTile + (int32_t) ceil((double) radius / tileWidth) + 1,
102  numTilesWidth);
103 
104  // Position of the lines / columns to copy from the file
105  _minRowFile = (uint32_t) std::max((int32_t) (_minRowCentralTile - radius),
106  (int32_t) 0);
107  _maxRowFile =
108  std::min((_indexRowCenterTile + 1) * tileHeight + radius, imageHeight);
109  _minColFile = (uint32_t) std::max((int32_t) (_minColCentralTile - radius),
110  (int32_t) 0);
111  _maxColFile =
112  std::min((_indexColCenterTile + 1) * tileWidth + radius, imageWidth);
113 
114  // Count of the lines / columns to copy from the file
115  _rowFilledFromFile = _maxRowFile - _minRowFile;
116  _colFilledFromFile = _maxColFile - _minColFile;
117 
118  // Number of pixels to fill with ghost values
119  _topFill = (int32_t) (_minRowCentralTile - radius) < 0 ? radius
120  - _minRowCentralTile : 0;
121  _leftFill = (int32_t) (_minColCentralTile - radius) < 0 ? radius
122  - _minColCentralTile : 0;
123  _bottomFill =
124  (_topFill + _rowFilledFromFile) < _viewHeight ? _viewHeight
125  - (_topFill + _rowFilledFromFile) : 0;
126  _rightFill =
127  (_leftFill + _colFilledFromFile) < _viewWidth ? _viewWidth
128  - (_leftFill + _colFilledFromFile) : 0;
129  _numberTilesToLoad = (_indexRowMaxTile - _indexRowMinTile)
130  * (_indexColMaxTile - _indexColMinTile);
131  }
