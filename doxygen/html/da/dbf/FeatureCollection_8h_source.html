98  : _imageWidth(0), _imageHeight(0) {

107  *analyseGraph; //< Analyse graph
108 
109  htgs::TaskGraphRuntime
110  *analyseRuntime; //< Analyse runtime
111 
112  assert(rank == 8 || rank == 4);
113  try {
114  //Create the Fast image and get the info from it
115  //The radius of 1 is used to make the link between the tiles
116  fi = new fi::FastImage<FileType>(tileLoader, 1);
117  fi->getFastImageOptions()->setNumberOfViewParallel(numberOfViewParallel);
118  fi->configureAndRun();
119  imageHeight = fi->getImageHeight();
120  imageWidth = fi->getImageWidth();
121 
122  //Set the number of threads if incorrect
123  if (numberOfThreadsParallel == 0) { numberOfThreadsParallel = 8; }

125  } catch (const fi::FastImageException &e) {
126  std::cerr << e.what() << std::endl;
127  exit(EXIT_FAILURE);

130  exit(EXIT_FAILURE);

133  // Create the analyse graph
134  analyseGraph = new htgs::TaskGraphConf<htgs::MemoryData<fi::View<FileType>>,
135  ListBlobs>;
136  auto viewAnalyseTask = new ViewAnalyser<FileType>(numberOfThreadsParallel,
137  fi,
138  rank,
139  background);
140  auto fileCreation = new BlobMerger(imageHeight,
141  imageWidth,
142  fi->getNumberTilesHeight()
143  * fi->getNumberTilesWidth());

145  analyseGraph->setGraphConsumerTask(viewAnalyseTask);
146  analyseGraph->addEdge(viewAnalyseTask, fileCreation);

161  }

168  }
169

174  fi->getImageHeight(0),
175  fi->getImageWidth(0));

366  this->end(),

368  std::sort(fc.begin(),
369  fc.end(),
370  [](Feature &a, Feature &b) { return a.getId() < b.getId(); });
371  for (auto a = this->begin(), b = fc.begin();
372  a != this->end() && b != fc.end(); ++a, b++) {
373  if ((*a) != (*b)) {
374  answer = false;
375  break;
376  }
377  }
378  } else { answer = false; }

385  void createLabeledMask(const std::string &pathLabeledMask,
386  const uint32_t tileSize = 1024) {
387  if ((tileSize & (tileSize - 1)) != 0) {
388  std::stringstream message;
389  message
390  << "Feature Collection ERROR: The tiling asked is not a power of 2.";
391  std::string m = message.str();
392  throw (fi::FastImageException(m));
393  }
394  // Get image size
395  auto
396  imageWidth = this->getImageWidth(),
397  imageHeight = this->getImageHeight();
398 
399  // Create two tiles pointer: one empty, one to be filled
400  std::vector<uint32_t>
401  *emptyTile = new std::vector<uint32_t>(tileSize * tileSize, 0),
402  *currentTile = nullptr;
403 
404  // Create a pair to create a coordinate (row/col)
405  std::pair<uint32_t, uint32_t>
406  currentCoordinates;
407 
408  // Create the tiff file
409  TIFF
410  *tif = TIFFOpen(pathLabeledMask.c_str(), "w");
411 
412  uint32_t
413  upperLeftRowFeature = 0,
414  upperLeftColFeature = 0,
415  bottomRightRowFeature = 0,
416  bottomRightColFeature = 0,
417  minRowTile = 0,
418  minColTile = 0,
419  upperLeftRow = 0,
420  upperLeftCol = 0,
421  bottomRightRow = 0,
422  bottomRightCol = 0,
423  indexMinTileRow = 0,
424  indexMaxTileRow = 0,
425  indexMinTileCol = 0,
426  indexMaxTileCol = 0;
427 
428  // Create a map of coordinate -> Tile
429  std::map<std::pair<uint32_t, uint32_t>, std::vector<uint32_t> *>
430  loadedTiles;
431 
432  if (tif != nullptr) {
433  TIFFSetField(tif, TIFFTAG_IMAGEWIDTH, imageWidth);
434  TIFFSetField(tif, TIFFTAG_IMAGELENGTH, imageHeight);
435  TIFFSetField(tif, TIFFTAG_TILELENGTH, tileSize);
436  TIFFSetField(tif, TIFFTAG_TILEWIDTH, tileSize);
437  TIFFSetField(tif, TIFFTAG_BITSPERSAMPLE, 8 * sizeof(uint32_t));
438  TIFFSetField(tif, TIFFTAG_ROWSPERSTRIP, 1);
439  TIFFSetField(tif, TIFFTAG_SAMPLEFORMAT, SAMPLEFORMAT_UINT);
440  TIFFSetField(tif, TIFFTAG_PLANARCONFIG, PLANARCONFIG_CONTIG);
441  TIFFSetField(tif, TIFFTAG_COMPRESSION, COMPRESSION_NONE);
442  TIFFSetField(tif, TIFFTAG_PHOTOMETRIC, PHOTOMETRIC_MINISBLACK);
443  TIFFSetField(tif, TIFFTAG_ORIENTATION, ORIENTATION_TOPLEFT);
444 
445  // For every feature in the feature collection
446  for (const auto &feature : this->getVectorFeatures()) {
447  // Get the bounding box
448  upperLeftRowFeature = feature.getBoundingBox().getUpperLeftRow();
449  upperLeftColFeature = feature.getBoundingBox().getUpperLeftCol();
450  bottomRightRowFeature = feature.getBoundingBox().getBottomRightRow();
451  bottomRightColFeature = feature.getBoundingBox().getBottomRightCol();
452 
453  indexMinTileRow = upperLeftRowFeature / tileSize;
454  indexMinTileCol = upperLeftColFeature / tileSize;
455 
456  indexMaxTileRow = (bottomRightRowFeature - 1) / tileSize;
457  indexMaxTileCol = (bottomRightColFeature - 1) / tileSize;
458 
459  // For every view in the bounding box
460  for (uint32_t indexRow = indexMinTileRow; indexRow <= indexMaxTileRow;
461  ++indexRow) {
462  for (uint32_t indexCol = indexMinTileCol; indexCol <= indexMaxTileCol;
463  ++indexCol) {
464 
465  // Get the tile, from the map or create one
466  currentCoordinates =
467  std::pair<uint32_t, uint32_t>(indexRow, indexCol);
468  if (loadedTiles.find(currentCoordinates) == loadedTiles.end()) {
469  loadedTiles[currentCoordinates] =
470  new std::vector<uint32_t>(tileSize * tileSize, 0);
471  }
472  currentTile = loadedTiles[currentCoordinates];
473 
474  // Get the indexes used for the computation
475  minRowTile = indexRow * tileSize;
476  minColTile = indexCol * tileSize;
477 
478  upperLeftRow = std::max(minRowTile, upperLeftRowFeature);
479  upperLeftCol = std::max(minColTile, upperLeftColFeature);
480 
481  bottomRightRow =
482  std::min(bottomRightRowFeature, (indexRow + 1) * tileSize);
483  bottomRightCol =
484  std::min(bottomRightColFeature, (indexCol + 1) * tileSize);
485 
486  // For every pixel in the tile
487  for (uint32_t rowTileFeature = upperLeftRow;
488  rowTileFeature < bottomRightRow; ++rowTileFeature) {
489  for (uint32_t colTileFeature = upperLeftCol;
490  colTileFeature < bottomRightCol; ++colTileFeature) {
491 
492  // If the pixel is in the feature
493  if (feature.isInBitMask(rowTileFeature, colTileFeature)) {
494  currentTile->at((rowTileFeature - minRowTile) * tileSize
495  + (colTileFeature - minColTile))
496  = (uint32_t) (feature.getId() + 1);
497  }
498  }
499  }
500  }
501  }
502  }
503 
504  indexMaxTileRow = (this->getImageHeight() - 1) / tileSize;
505  indexMaxTileCol = (this->getImageWidth() - 1) / tileSize;
506 
507  // For every tile in the image
508  for (uint32_t indexRow = 0; indexRow <= indexMaxTileRow; indexRow++) {
509  for (uint32_t indexCol = 0; indexCol <= indexMaxTileCol; indexCol++) {
510 
511  // Get the tile from it coordinates in the map
512  currentCoordinates =
513  std::pair<uint32_t, uint32_t>(indexRow, indexCol);
514  // If it doesn't exist write the blank tile
515  if (loadedTiles.find(currentCoordinates) == loadedTiles.end()) {
516  TIFFWriteTile(tif,
517  emptyTile,
518  indexCol * tileSize,
519  indexRow * tileSize,
520  0,
521  0);
522  } else {
523  // Write the filled tile
524  TIFFWriteTile(tif,
525  loadedTiles[currentCoordinates]->data(),
526  indexCol * tileSize,
527  indexRow * tileSize,
528  0,
529  0);
530  }
531 
532  }
533  }
534  // Close the tif
535  TIFFClose(tif);
536  } else {
537  std::cerr << "The File " << pathLabeledMask << " can't be opened."
538  << std::endl;
539  exit(1);
540  }
541 
542  delete emptyTile;
543  for (auto &tile : loadedTiles) {
544  delete tile.second;
545  }
546  }
