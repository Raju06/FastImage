109  void initCache(uint32_t numTilesHeight,
110  uint32_t numTilesWidth,
111  uint32_t tileHeight,
112  uint32_t tileWidth) {
113 
114  uint32_t nbTilesInImage = numTilesHeight * numTilesWidth;
115 
116  _numTilesHeight = numTilesHeight;
117  _numTilesWidth = numTilesWidth;
118 
119  // If the number of tiles to be cached has been set to 0 (default value),
120  // set the number to 2 * number of tiles in a row
121  if (_nbTilesCache == 0) { _nbTilesCache = 2 * numTilesWidth; }
122 
123  // If the number of tiles to be cached is superior to the number of tiles in
124  // the image, set it to the number of tiles in the image
125  if (nbTilesInImage < _nbTilesCache) { _nbTilesCache = nbTilesInImage; }
126 
127  // Create the matrix
128  for (uint32_t row = 0; row < numTilesHeight; ++row) {
129  std::vector<CachedTileType> tempV;
130  for (uint32_t col = 0; col < numTilesWidth; ++col) {
131  tempV.push_back(nullptr);
132  }
133  _mapCache.push_back(tempV);
134  }
135 
136  // Fill the pool
137  for (uint32_t tileCnt = 0; tileCnt < _nbTilesCache; ++tileCnt) {
138  _pool.push(new CachedTile<UserType>(tileWidth, tileHeight));
139  }
140  };

156  CachedTileType getLockedTile(uint32_t indexRow, uint32_t indexCol) {
157  if (!(indexRow >= 0 && indexRow < _numTilesHeight && indexCol >= 0
158  && indexCol < _numTilesWidth)) {
159  std::stringstream message;
160  message << "Tile Loader ERROR: The index is not correct: ("
161  << (int) indexRow << ", "
162  << (int) indexCol << ")";
163  std::string m = message.str();
164  throw (FastImageException(m));
165  }
166  CachedTileType tile;
167 
168  this->lock();
169  auto begin = std::chrono::high_resolution_clock::now();
170 
171  if (isInCache(indexRow, indexCol)) {
172  // Tile is in cache
173  _hit += 1;
174  tile = getCachedLockedTile(indexRow, indexCol);
175  } else {
176  // Tile is not in the cache
177  _miss += 1;
178  if (_pool.empty()) { recycleTile(); }
179  tile = getNewLockedTile(indexRow, indexCol);
180  }
181  auto end = std::chrono::high_resolution_clock::now();
182  _timeGet += std::chrono::duration_cast<std::chrono::nanoseconds>(
183  end - begin).count();
184  this->unlock();
185 
186  return tile;
187  }

287  void recycleTile() {
288  CachedTileType toRecycle;
289 
290  auto begin = std::chrono::high_resolution_clock::now();
291 
292  // Get LRU Tile
293  toRecycle = _lru.back();
294  toRecycle->lock();
295  _lru.pop_back();
296 
297  // Clean The Tile
298  _mapLRU.erase(toRecycle);
299  _mapCache[toRecycle->getIndexRowGlobal()][toRecycle->getIndexColGlobal()] =
300  nullptr;
301  toRecycle->setIndexRowGlobal(0);
302  toRecycle->setIndexColGlobal(0);
303  toRecycle->setNewTile(true);
304 
305  // Put it back in the pool
306  _pool.push(toRecycle);
307 
308  auto end = std::chrono::high_resolution_clock::now();
309  _timeRecycle += std::chrono::duration_cast<std::chrono::nanoseconds>(
310  end - begin).count();
311 
312  toRecycle->unlock();
313  }

319  CachedTileType getNewLockedTile(uint32_t indexRow, uint32_t indexCol) {
320  // Get tile from the pool
321  CachedTileType tile = _pool.front();
322  tile->lock();
323  _pool.pop();
324 
325  // Set tile information except data
326  tile->setIndexColGlobal(indexCol);
327  tile->setIndexRowGlobal(indexRow);
328 
329  // Register the tile

331  _lru.push_front(tile);

340  CachedTileType getCachedLockedTile(uint32_t indexRow, uint32_t indexCol) {
341  assert(_mapCache[indexRow][indexCol] != nullptr);
342 
343  // Get the tile
344  CachedTileType tile = _mapCache[indexRow][indexCol];
345  tile->lock();
346 
347  // Update the tile position in the LRU
348  _lru.erase(_mapLRU[tile]);
349  _lru.push_front(tile);
350  _mapLRU[tile] = _lru.begin();
351 
352  return tile;
