93  void analyseNeighbour4(int32_t row, int32_t col) {
94  // Top Pixel
95  if (row >= 1) {
96  if (needVisit(row - 1, col)) {
97  _toVisit.emplace(row - 1, col);
98  }
99  }
100  // Bottom pixel
101  if (row + 1 < _tileHeight) {
102  if (needVisit((row + 1), col)) {
103  _toVisit.emplace(row + 1, col);
104  }
105  }
106  // Left pixel
107  if (col >= 1) {
108  if (needVisit(row, col - 1)) {
109  _toVisit.emplace(row, col - 1);
110  }
111  }
112  // Right pixel
113  if (col + 1 < _tileWidth) {
114  if (needVisit(row, col + 1)) {
115  _toVisit.emplace(row, col + 1);
116  }
117  }
118  // Add a pixel to to merge if the bottom pixel is outside of the tile, but
119  // had a foreground value
120  if (row + 1 == _tileHeight
121  && row + _view->getGlobalYOffset() + 1 != _imageHeight) {
122  if (_view->getPixel(row + 1, col) != _background) {
123  _vAnalyse->addToMerge(_currentBlob,
124  std::pair<int32_t, int32_t>(
125  row + 1 + _view->getGlobalYOffset(),
126  col + _view->getGlobalXOffset()));
127  }
128  }
129  // Add a pixel to to merge if the right pixel is outside of the tile, but
130  // had a foreground value
131  if (col + 1 == _tileWidth
132  && col + _view->getGlobalXOffset() + 1 != _imageWidth) {
133  if (_view->getPixel(row, col + 1) != _background) {
134  _vAnalyse->addToMerge(_currentBlob,
135  std::pair<int32_t, int32_t>(
136  row + _view->getGlobalYOffset(),
137  col + 1 + _view->getGlobalXOffset()));
138  }
139  }

146  void analyseNeighbour8(int32_t row, int32_t col) {
147  int32_t
148  minRow = std::max(0, row - 1),
149  maxRow = std::min(_view->getTileHeight(), row + 2),
150  minCol = std::max(0, col - 1),
151  maxCol = std::min(_view->getTileWidth(), col + 2);
152 
153  // Analyse the pixels around the pixel
154  for (int32_t rowP = minRow; rowP < maxRow; ++rowP) {
155  for (int32_t colP = minCol; colP < maxCol; ++colP) {
156  if (needVisit(rowP, colP)) {
157  _toVisit.emplace(rowP, colP);
158  }
159  }
160  }
161 
162  // Add a pixel to to merge if the bottom pixel is outside of the tile,
163  // but had a foreground value
164  if (row + 1 == _tileHeight
165  && row + _view->getGlobalYOffset() + 1 != _imageHeight) {
166  if (_view->getPixel(row + 1, col) != _background) {
167  _vAnalyse->addToMerge(_currentBlob,
168  std::pair<int32_t, int32_t>(
169  row + 1 + _view->getGlobalYOffset(),
170  col + _view->getGlobalXOffset())
171  );
172  }
173  }
174 
175  // Add a pixel to to merge if the right pixel is outside of the tile, but
176  // had a foreground value
177  if (col + 1 == _tileWidth
178  && col + _view->getGlobalXOffset() + 1 != _imageWidth) {
179  if (_view->getPixel(row, col + 1) != _background) {
180  _vAnalyse->addToMerge(_currentBlob,
181  std::pair<int32_t, int32_t>(
182  row + _view->getGlobalYOffset(),
183  col + 1 + _view->getGlobalXOffset()));
184  }
185  }
186 
187  // Add a pixel to to merge if the bottom-right pixel is outside of the tile,
188  // but had a foreground value
189  if (row + 1 == _tileHeight
190  && row + _view->getGlobalYOffset() + 1 != _imageHeight
191  && col + 1 == _tileWidth
192  && col + _view->getGlobalXOffset() + 1 != _imageWidth) {
193  if (_view->getPixel(row + 1, col + 1) != _background) {
194  _vAnalyse->addToMerge(_currentBlob,
195  std::pair<int32_t, int32_t>(
196  row + 1 + _view->getGlobalYOffset(),
197  col + 1 + _view->getGlobalXOffset()));
198  }
199  }
200 
201  // Add a pixel to to merge if the top-right pixel is outside of the tile,
202  // but had a foreground value
203  if (row == 0 && row + _view->getGlobalYOffset() > 0 && col + 1 == _tileWidth
204  && col + _view->getGlobalXOffset() + 1 != _imageWidth) {
205  if (_view->getPixel(row - 1, col + 1) != _background) {
206  _vAnalyse->addToMerge(_currentBlob,
207  std::pair<int32_t, int32_t>(
208  row - 1 + _view->getGlobalYOffset(),
209  col + 1 + _view->getGlobalXOffset()));
210  }

216  void executeTask(std::shared_ptr<MemoryData<fi::View<UserType>>> view)
217  override {
218  _view = view->get();
219 
220  //Set tup the environment for the tile
221  _tileHeight = _view->getTileHeight();
222  _tileWidth = _view->getTileWidth();
223  _vAnalyse = new ViewAnalyse();
224  _toVisit.clear();
225  _currentBlob = nullptr;
226 
227  // For every pixel
228  for (int32_t row = 0; row < _tileHeight; ++row) {
229  for (int32_t col = 0; col < _tileWidth; ++col) {
230 
231  // Analyse the neighbour if need be
232  while (!_toVisit.empty()) {
233  auto neighbourCoord = *_toVisit.begin();
234  _toVisit.erase(_toVisit.begin());
235  _view->setPixel(neighbourCoord.first,
236  neighbourCoord.second,
237  _background);
238  _currentBlob->addPixel(
239  _view->getGlobalYOffset() + neighbourCoord.first,
240  _view->getGlobalXOffset() + neighbourCoord.second);
241  if (_rank == 4) {
242  analyseNeighbour4(neighbourCoord.first, neighbourCoord.second);
243  } else {
244  analyseNeighbour8(neighbourCoord.first, neighbourCoord.second);
245  }
246  }
247 
248  // Save the current blob if need be
249  if (_currentBlob != nullptr) {
250  _vAnalyse->insertBlob(_currentBlob);
251  _currentBlob = nullptr;
252  }
253 
254  // Test if the current pixel need to be visited
255  if (needVisit(row, col)) {
256  //Create a blob for it
257  _currentBlob = new Blob();
258  _currentBlob->addPixel(_view->getGlobalYOffset() + row,
259  _view->getGlobalXOffset() + col);
260  _view->setPixel(row, col, _background);
261  // Analyse is neighbour
262  if (_rank == 4) {
263  analyseNeighbour4(row, col);
264  } else {
265  analyseNeighbour8(row, col);
266  }
267  }
268  }
269  }
270  // Save the current blob if the BR pixel is alone and is a blob by itself
271  if (_currentBlob != nullptr) {
272  _vAnalyse->insertBlob(_currentBlob);
273  _currentBlob = nullptr;
274  }
275 
276  // Release the view memory
277  view->releaseMemory();
278  // Add the analyse
279  this->addResult(_vAnalyse);
280  }
