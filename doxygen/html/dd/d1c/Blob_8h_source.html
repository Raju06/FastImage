186  Blob *mergeAndDelete(Blob *blob) {
187  Blob
188  *toDelete = nullptr,
189  *destination = nullptr;
190 
191  // Decide which one (between this and blob) will be deleted
192  if (this->getCount() >= blob->getCount()) {
193  destination = this;
194  toDelete = blob;
195  } else {
196  destination = blob;
197  toDelete = this;
198  }
199 
200  // Set blob metadata
201  destination->setRowMin(
202  std::min(destination->getRowMin(), toDelete->getRowMin()));
203  destination->setColMin(
204  std::min(destination->getColMin(), toDelete->getColMin()));
205  destination->setRowMax(
206  std::max(destination->getRowMax(), toDelete->getRowMax()));
207  destination->setColMax(
208  std::max(destination->getColMax(), toDelete->getColMax()));
209 
210  // Merge sparse matrix
211  for (auto rowCol : toDelete->getRowCols()) {
212  for (auto col : rowCol.second) {
213  destination->addRowCol(rowCol.first, col);
214  }
215  }
216 
217  // update pixel count
218  destination->setCount(toDelete->getCount() + destination->getCount());
219 
220  // Delete unused Blob
221  delete (toDelete);
222  // Return merged blob
223  return destination;
224  }
