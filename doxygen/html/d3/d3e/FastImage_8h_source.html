329  void configureAndRun() {
330  if (!_hasBeenConfigured) {
331  assert(_fastImageOptions->getNumberOfViewParallel() > 0);
332  _hasBeenConfigured = true;
333  if (_fastImageOptions->getNumberOfTileLoader() == 0)
334  _fastImageOptions->setNumberOfTileLoader(1);
335 
336  ViewLoader<UserType> *viewLoader = nullptr;
337  _viewCounter = nullptr;
338 
339  std::vector<size_t> numViewsParallel;
340  std::vector<std::shared_ptr<htgs::IMemoryAllocator<View<UserType>>>>
341  viewAllocators;
342 
343  for (uint32_t level = 0; level < _tileLoader->getNbPyramidLevels();
344  level++) {
345  // Create the cache
346  auto cache =
347  new FigCache<UserType>(
348  this->_fastImageOptions->getNumberOfTilesToCache()
349  );
350  // Init the cache
351  cache->initCache(this->getNumberTilesHeight(level),
352  this->getNumberTilesWidth(level),
353  this->getTileHeight(level),
354  this->getTileWidth(level));
355  _allCache.push_back(cache);
356 
357  size_t
358  numViewParallelTemp = _fastImageOptions->getNumberOfViewParallel();
359 
360  // Set the number of view parallel
361  if (getNumberTilesWidth(level) * getNumberTilesHeight(level)
362  < _fastImageOptions->getNumberOfViewParallel()) {
363  numViewParallelTemp =
364  getNumberTilesWidth(level) * getNumberTilesHeight(level);
365  }
366 
367  numViewsParallel.push_back(numViewParallelTemp);
368  auto viewAllocator =
369  std::shared_ptr<ViewAllocator<UserType>>(
370  new ViewAllocator<UserType>(
371  getViewHeight(level),
372  getViewWidth(level)
373  )
374  );
375  viewAllocators.push_back(viewAllocator);
376  }
377 
378  auto memManager =
379  new VariableMemoryManager<View<UserType>>("viewMem",
380  numViewsParallel,
381  viewAllocators,
382  htgs::MMType::Static);
383 
384  // Create the Fast Image graph
385  _taskGraph = new htgs::TaskGraphConf<HTGSViewRequestData<UserType>,
386  htgs::MemoryData<View<UserType>>>();
387 
388  // Set the cache
389  _tileLoader->setCache(_allCache);
390 
391  // Init the graph's parts
392  viewLoader =
393  new ViewLoader<UserType>(
394  this->_fastImageOptions->getNbReleasePyramid()
395  );
396  _viewCounter =
397  new ViewCounter<UserType>(_fastImageOptions->getFillingType(),
398  _fastImageOptions->isOrderPreserved());
399 
400  if (this->getNbPyramidLevels() == 1) {
401  // Set graph parts
402  _taskGraph->setGraphConsumerTask(viewLoader);
403  _taskGraph->addEdge(viewLoader, _tileLoader);
404  _taskGraph->addEdge(_tileLoader, _viewCounter);
405  _taskGraph->addGraphProducerTask(_viewCounter);
406 
407  _taskGraph->addCustomMemoryManagerEdge(viewLoader, memManager);
408  } else {
409  auto pyramidGraph =
410  new htgs::TaskGraphConf<HTGSViewRequestData<UserType>,
411  htgs::MemoryData<View<UserType>>>();
412  pyramidGraph->setGraphConsumerTask(viewLoader);
413  pyramidGraph->addEdge(viewLoader, _tileLoader);
414  pyramidGraph->addEdge(_tileLoader, _viewCounter);
415  pyramidGraph->addGraphProducerTask(_viewCounter);
416  pyramidGraph->addCustomMemoryManagerEdge(viewLoader, memManager);
417  auto
418  execPipeline =
419  new htgs::ExecutionPipeline<HTGSViewRequestData<UserType>,
420  htgs::MemoryData<View<UserType>>>(
421  this->getNbPyramidLevels(), pyramidGraph);
422  auto distributeRule = new DistributePyramidRule<UserType>();
423  execPipeline->addInputRule(distributeRule);
424  _taskGraph->setGraphConsumerTask(execPipeline);
425  _taskGraph->addGraphProducerTask(execPipeline);
426  }
427  // Launch the graph
428  _runtime = new htgs::TaskGraphRuntime(_taskGraph);
429  _runtime->executeRuntime();
430  }

589  void requestTile(uint32_t rowIndex,
590  uint32_t colIndex,
591  bool finishRequestingTiles,
592  uint32_t level = 0) {
593  assert(rowIndex < this->getNumberTilesHeight());
594  assert(colIndex < this->getNumberTilesWidth());
595  assert(_hasBeenConfigured);
596 
597  if (this->isFinishedRequestingViews()) {
598  return;
599  }
600 
601  std::queue<std::pair<uint32_t, uint32_t>> fifo;
602  fifo.push(std::make_pair(rowIndex, colIndex));
603  _viewCounter->addTraversal(fifo);
604  sendRequest(rowIndex, colIndex, level);
605  if (finishRequestingTiles) {
606  this->finishedRequestingTiles();
607  }
608  }

615  void requestFeature(const fc::Feature &feature, uint32_t level = 0) {
616  assert(_hasBeenConfigured);
617  const fc::BoundingBox &bB = feature.getBoundingBox();
618 
619  uint32_t
620  indexRowMin = bB.getUpperLeftRow() / (this->getTileHeight()),
621  indexRowMax = 0,
622  indexColMin = bB.getUpperLeftCol() / this->getTileWidth(),
623  indexColMax = 0;
624 
625  // Handle border case
626  if (bB.getBottomRightCol() == this->getImageWidth())
627  indexColMax = this->getNumberTilesWidth();
628  else
629  indexColMax = (bB.getBottomRightCol() / this->getTileWidth()) + 1;
630  if (bB.getBottomRightRow() == this->getImageHeight())
631  indexRowMax = this->getNumberTilesHeight();
632  else
633  indexRowMax = (bB.getBottomRightRow() / this->getTileHeight()) + 1;
634 
635  this->setNumberTilesFeatureComputed(0);
636  this->setNumberTilesFeatureTotal(
637  (indexRowMax - indexRowMin) * (indexColMax - indexColMin));
638 
639  std::queue<std::pair<uint32_t, uint32_t>> fifo;
640 
641  for (auto indexRow = indexRowMin; indexRow < indexRowMax; ++indexRow) {
642  for (auto indexCol = indexColMin; indexCol < indexColMax; ++indexCol) {
643  fifo.push(std::make_pair(indexRow, indexCol));
644  }
645  }
646 
647  _viewCounter->addTraversal(fifo);
648 
649  for (auto indexRow = indexRowMin; indexRow < indexRowMax; ++indexRow) {
650  for (auto indexCol = indexColMin; indexCol < indexColMax; ++indexCol) {
651  sendRequest(indexRow, indexCol, level);
652  }
653  }
654  }

660  void requestAllTiles(bool finishRequestingTiles, uint32_t level = 0) {
661  assert(_hasBeenConfigured);
662  if (this->isFinishedRequestingViews())
663  return;
664 
665  Traversal traversal = Traversal(this->_fastImageOptions->getTraversalType(),
666  getNumberTilesHeight(level),
667  getNumberTilesWidth(level));
668 
669  _viewCounter->addTraversal(traversal.getQueue());
670 
671  for (auto step : traversal.getTraversal()) {
672  sendRequest(step.first, step.second, level);
673  }
674 
675  if (finishRequestingTiles) {
676  this->finishedRequestingTiles();
677  }
678  }
